```js
  import { createApp } from "vue";
  import App from "./App.vue";

  import Vuesax from "vuesax-alpha";
  // vuesax style
  import "vuesax-alpha/dist/vuesax.css";

  const app = createApp(App);
  app.use(Vuesax);
  app.mount("#app");
```
